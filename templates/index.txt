  <script>
    const video = document.getElementById('webcam');
    const startBtn = document.getElementById('start-btn');
    const emotionBox = document.getElementById('emotion-box');

    startBtn.addEventListener('click', async () => {
      try {
        const stream = await navigator.mediaDevices.getUserMedia({ video: true });
        video.srcObject = stream;
        emotionBox.textContent = "Camera started... detecting mood!";
        emotionBox.classList.add('detected');
      } catch (error) {
        emotionBox.textContent = "Error accessing webcam ðŸ˜”";
        emotionBox.style.color = "red";
      }
    });
